129 inline bool IsAddressDefined(const Locator_t& loc)
130 {
131  if (loc.kind == LOCATOR_KIND_UDPv4 || loc.kind == LOCATOR_KIND_TCPv4) // WAN addr in TCPv4 is optional, isn't?
132  {
133  for (uint8_t i = 12; i < 16; ++i)
134  {
135  if (loc.address[i] != 0)
136  return true;
137  }
138  }
139  else if (loc.kind == LOCATOR_KIND_UDPv6 || loc.kind == LOCATOR_KIND_TCPv6)
140  {
141  for (uint8_t i = 0; i < 16; ++i)
142  {
143  if (loc.address[i] != 0)
144  return true;
145  }
146  }
147  return false;
148 }

160 inline bool operator==(const Locator_t&loc1, const Locator_t& loc2)
161 {

163  return false;
164  if (loc1.port != loc2.port)
