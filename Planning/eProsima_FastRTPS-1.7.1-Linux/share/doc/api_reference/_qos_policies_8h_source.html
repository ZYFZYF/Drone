102  inline rtps::DurabilityKind_t durabilityKind() const
103  {
104  switch (kind)
105  {
106  default:
107  case VOLATILE_DURABILITY_QOS: return rtps::VOLATILE;
108  case TRANSIENT_LOCAL_DURABILITY_QOS: return rtps::TRANSIENT_LOCAL;
109  case TRANSIENT_DURABILITY_QOS: return rtps::TRANSIENT;
110  case PERSISTENT_DURABILITY_QOS: return rtps::PERSISTENT;
111  }
112  }
113 
114  bool operator==(const DurabilityQosPolicy& b) const
115  {
116  return (this->kind == b.kind) &&
117  Parameter_t::operator==(b) &&
118  QosPolicy::operator==(b);
119  }
120 
124  inline void durabilityKind(const rtps::DurabilityKind_t new_kind)
125  {
126  switch (new_kind)
127  {
128  default:
129  case rtps::VOLATILE: kind = VOLATILE_DURABILITY_QOS; break;
130  case rtps::TRANSIENT_LOCAL: kind = TRANSIENT_LOCAL_DURABILITY_QOS; break;
131  case rtps::TRANSIENT: kind = TRANSIENT_DURABILITY_QOS; break;
132  case rtps::PERSISTENT: kind = PERSISTENT_DURABILITY_QOS; break;
133  }
134 
135  }
136 
142  bool addToCDRMessage(rtps::CDRMessage_t* msg) override;
143 
144 public:
145  DurabilityQosPolicyKind_t kind;
