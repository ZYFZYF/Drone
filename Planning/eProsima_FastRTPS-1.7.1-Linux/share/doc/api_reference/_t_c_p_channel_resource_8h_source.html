188  inline bool ChangeStatus(eConnectionStatus s)
189  {
190  std::unique_lock<std::mutex> scoped(mStatusMutex);
191  if (mConnectionStatus != s)
192  {
193  mConnectionStatus = s;
194  if (mConnectionStatus == eEstablished)
195  {
196  SendPendingOpenLogicalPorts();
197  }
198  mNegotiationCondition.notify_all();
199  return true;
